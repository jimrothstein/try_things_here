sprintf has some nice features!

sprintf("hello %s", "jim")
      ## [1] "hello jim"

sprintf("hello %s", 23)
      ## [1] "hello 23"

sprintf("hello %04s", 23)         # min of 4
      ## [1] "hello   23"
sprintf("hello %04f", 23)         # 23.000000
      ## [1] "hello 23.000000"

sprintf("hello %04i", 23)         # int, min of 4 digits
      ## [1] "hello 0023"
