one_sample = function(DUMMY = NULL) {
    sample = rnorm(n=n,mean= m, sd=sigma )
    #hist(sample)

    avg = mean(sample)

    sd = sd(sample)
    L = avg - 1.96*sd/sqrt(n)
    U = avg + 1.96*sd/sqrt(n)

    contains_TRUE =  (L < m)   & ( m < U )
    new  <- tibble::tibble(avg=avg, L=L, U=U, contains_TRUE = contains_TRUE)
    t <<- rbind(t, new)
    invisible(t)
}

t  <- tibble::tibble()
